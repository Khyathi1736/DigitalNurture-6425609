# 1. Create and enter the project folder
mkdir git-hands-on-1
cd git-hands-on-1

# 2. Initialize Git
git init

# 3. Create a README file
echo "# Git Hands-on 1" > readme.md

# 4. Stage the file
git add readme.md

# 5. Commit the file
git commit -m "Initial commit: Added README"

# 6. Create a new repository on GitHub (do this on GitHub website)
#    Copy the repository URL (e.g., https://github.com/username/git-hands-on-1.git)

# 7. Link local repo to GitHub
git remote add origin <YOUR_GITHUB_REPO_URL>

# 8. Push to GitHub
git branch -M main
git push -u origin main
